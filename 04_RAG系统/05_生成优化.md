## 上下文压缩

### 为什么需要压缩？
```
问题：
1. 检索到的文档可能包含大量无关信息
2. 上下文窗口有限
3. 噪声信息影响生成质量

解决：
1. 提取关键信息
2. 去除冗余内容
3. 保留相关信息
```

```python
def extract_relevant片段(documents, query, max_length=200):
    """提取文档中的相关片段"""
    
    relevant_fragments = []
    
    for doc in documents:
        # 分句
        sentences = doc.page_content.split('。')
        
        # 计算每句与查询的相关性
        scored_sentences = []
        for sentence in sentences:
            if sentence.strip():
                score = calculate_similarity(sentence, query)
                scored_sentences.append((score, sentence))
        
        # 选择最相关的句子
        scored_sentences.sort(reverse=True)
        top_sentences = [s for _, s in scored_sentences[:3]]
        
        # 合并
        fragment = '。'.join(top_sentences)
        relevant_fragments.append(fragment)
    
    return relevant_fragments
```

#### 2. 引用后处理
```python
def parse_citations(response, documents):
    """解析引用并验证"""
    
    # 提取引用标记
    import re
    citations = re.findall(r'\[(\d+)\]', response)
    
    # 验证引用
    valid_citations = []
    for cite in citations:
        idx = int(cite) - 1
        if 0 <= idx < len(documents):
            valid_citations.append({
                "marker": f"[{cite}]",
                "document": documents[idx]
            })
    
    return valid_citations
```

### 验证方法
```python
def verify_answer(answer, documents):
    """验证回答是否基于文档"""
    
    # 方法1：检查关键事实
    facts = extract_facts(answer)
    
    supported_facts = []
    unsupported_facts = []
    
    for fact in facts:
        is_supported = False
        for doc in documents:
            if fact in doc.page_content:
                is_supported = True
                break
        
        if is_supported:
            supported_facts.append(fact)
        else:
            unsupported_facts.append(fact)
    
    # 计算支持率
    support_rate = len(supported_facts) / len(facts) if facts else 0
    
    return {
        "support_rate": support_rate,
        "supported_facts": supported_facts,
        "unsupported_facts": unsupported_facts
    }
```

### 置信度评分
```python
def calculate_confidence(answer, documents, query):
    """计算回答的置信度"""
    
    scores = []
    
    # 1. 相关性分数
    relevance_score = calculate_relevance(answer, documents)
    scores.append(relevance_score * 0.4)
    
    # 2. 完整性分数
    completeness_score = calculate_completeness(answer, query)
    scores.append(completeness_score * 0.3)
    
    # 3. 一致性分数
    consistency_score = calculate_consistency(answer, documents)
    scores.append(consistency_score * 0.3)
    
    return sum(scores)
```

### 上下文管理
```python
class ConversationManager:
    """对话上下文管理"""
    
    def __init__(self, max_history=5):
        self.history = []
        self.max_history = max_history
    
    def add_interaction(self, question, answer):
        """添加对话记录"""
        self.history.append({
            "question": question,
            "answer": answer
        })
        
        # 保持历史长度
        if len(self.history) > self.max_history:
            self.history = self.history[-self.max_history:]
    
    def get_context(self):
        """获取对话上下文"""
        context = ""
        for item in self.history:
            context += f"用户：{item['question']}\n助手：{item['answer']}\n\n"
        return context
```

## 性能监控

### 监控指标
```python
class RAGMonitor:
    """RAG系统监控"""
    
    def __init__(self):
        self.metrics = {
            "total_queries": 0,
            "avg_retrieval_time": 0,
            "avg_generation_time": 0,
            "avg_confidence": 0
        }
    
    def log_query(self, query, retrieval_time, generation_time, confidence):
        """记录查询"""
        self.metrics["total_queries"] += 1
        
        # 更新平均值
        n = self.metrics["total_queries"]
        self.metrics["avg_retrieval_time"] = (
            self.metrics["avg_retrieval_time"] * (n-1) + retrieval_time
        ) / n
        
        self.metrics["avg_generation_time"] = (
            self.metrics["avg_generation_time"] * (n-1) + generation_time
        ) / n
        
        self.metrics["avg_confidence"] = (
            self.metrics["avg_confidence"] * (n-1) + confidence
        ) / n
```
