### 完整代码
```python
import os
from typing import List, Dict
from langchain.document_loaders import DirectoryLoader, PyPDFLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import FAISS
from langchain.llms import OpenAI
from langchain.chains import RetrievalQA
from langchain.prompts import PromptTemplate

class KnowledgeBaseQA:
    """知识库问答系统"""
    
    def __init__(self, docs_path: str, index_path: str = "faiss_index"):
        """
        初始化系统
        
        Args:
            docs_path: 文档目录路径
            index_path: 索引保存路径
        """
        self.docs_path = docs_path
        self.index_path = index_path
        
        # 初始化组件
        self.embeddings = OpenAIEmbeddings()
        self.llm = OpenAI(temperature=0)
        
        # 加载或创建向量索引
        if os.path.exists(index_path):
            self.vectorstore = self.load_index()
        else:
            self.vectorstore = self.build_index()
        
        # 创建QA链
        self.qa_chain = self.create_qa_chain()
    
    def load_documents(self) -> List:
        """加载文档"""
        print(f"正在加载文档从: {self.docs_path}")
        
        # 加载PDF
        loader = DirectoryLoader(
            self.docs_path,
            glob="**/*.pdf",
            loader_cls=PyPDFLoader
        )
        
        # 加载文本文件
        text_loader = DirectoryLoader(
            self.docs_path,
            glob="**/*.txt"
        )
        
        documents = loader.load() + text_loader.load()
        print(f"加载了 {len(documents)} 个文档")
        
        return documents
    
    def split_documents(self, documents: List) -> List:
        """分割文档"""
        text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=500,
            chunk_overlap=50,
            separators=["\n\n", "\n", " ", ""]
        )
        
        chunks = text_splitter.split_documents(documents)
        print(f"分割成 {len(chunks)} 个文本块")
        
        return chunks
    
    def build_index(self):
        """构建向量索引"""
        # 加载文档
        documents = self.load_documents()
        
        # 分割文档
        chunks = self.split_documents(documents)
        
        # 创建向量索引
        print("正在创建向量索引...")
        vectorstore = FAISS.from_documents(chunks, self.embeddings)
        
        # 保存索引
        vectorstore.save_local(self.index_path)
        print(f"索引已保存到: {self.index_path}")
        
        return vectorstore
    
    def load_index(self):
        """加载向量索引"""
        print(f"正在加载索引从: {self.index_path}")
        return FAISS.load_local(self.index_path, self.embeddings)
    
    def create_qa_chain(self):
        """创建问答链"""
        # 自定义提示模板
        prompt_template = """
        你是一个专业的助手。请基于以下上下文信息回答问题。
        
        要求：
        1. 只使用提供的上下文信息
        2. 如果上下文没有相关信息，请说明
        3. 回答要准确、完整
        
        上下文：
        {context}
        
        问题：{question}
        
        请回答：
        """
        
        PROMPT = PromptTemplate(
            template=prompt_template,
            input_variables=["context", "question"]
        )
        
        # 创建QA链
        qa_chain = RetrievalQA.from_chain_type(
            llm=self.llm,
            chain_type="stuff",
            retriever=self.vectorstore.as_retriever(
                search_kwargs={"k": 3}
            ),
            chain_type_kwargs={"prompt": PROMPT},
            return_source_documents=True
        )
        
        return qa_chain
    
    def query(self, question: str) -> Dict:
        """
        查询问题
        
        Args:
            question: 用户问题
            
        Returns:
            包含答案和来源的字典
        """
        result = self.qa_chain({"query": question})
        
        return {
            "answer": result["result"],
            "sources": [doc.metadata.get("source", "unknown") 
                       for doc in result["source_documents"]]
        }
    
    def update_knowledge_base(self):
        """更新知识库"""
        # 删除旧索引
        if os.path.exists(self.index_path):
            import shutil
            shutil.rmtree(self.index_path)
        
        # 重建索引
        self.vectorstore = self.build_index()
        self.qa_chain = self.create_qa_chain()
        print("知识库已更新")

# 使用示例
if __name__ == "__main__":
    # 初始化系统
    qa_system = KnowledgeBaseQA(
        docs_path="./company_docs",
        index_path="./kb_index"
    )
    
    # 查询
    result = qa_system.query("公司的休假政策是什么？")
    print(f"答案: {result['answer']}")
    print(f"来源: {result['sources']}")
```

### 项目概述

```
技术：RAG + 长上下文 + 多模态
```

### 核心功能
```python
class DocumentAssistant:
    """文档助手"""
    
    def __init__(self, llm, vectorstore):
        self.llm = llm
        self.vectorstore = vectorstore
    
    def summarize_document(self, document_path: str) -> str:
        """文档总结"""
        # 加载文档
        with open(document_path, 'r') as f:
            content = f.read()
        
        # 生成总结
        prompt = f"""
        请对以下文档进行总结，要求：
        1. 提取主要观点
        2. 保持逻辑结构
        3. 控制在300字以内
        
        文档内容：
        {content[:5000]}  # 限制长度
        
        总结：
        """
        
        return self.llm.generate(prompt)
    
    def extract_information(self, document_path: str, 
                          info_type: str) -> List:
        """提取特定信息"""
        
        # 检索相关段落
        retriever = self.vectorstore.as_retriever()
        
        if info_type == "dates":
            query = "文档中提到的日期、时间"
        elif info_type == "names":
            query = "文档中提到的人名、机构名"
        elif info_type == "numbers":
            query = "文档中提到的数字、数据"
        else:
            query = info_type
        
        docs = retriever.get_relevant_documents(query)
        
        # 提取信息
        information = []
        for doc in docs:
            # 使用正则表达式提取
            if info_type == "dates":
                import re
                dates = re.findall(r'\d{4}年\d{1,2}月\d{1,2}日', doc.page_content)
                information.extend(dates)
        
        return information
    
    def compare_documents(self, doc1_path: str, doc2_path: str) -> Dict:
        """文档对比"""
        
        # 加载文档
        with open(doc1_path, 'r') as f:
            content1 = f.read()
        with open(doc2_path, 'r') as f:
            content2 = f.read()
        
        # 对比分析
        prompt = f"""
        请对比分析以下两个文档，找出：
        1. 相同点
        2. 不同点
        3. 关键差异
        
        文档1：
        {content1[:3000]}
        
        文档2：
        {content2[:3000]}
        
        分析结果：
        """
        
        return self.llm.generate(prompt)
```

### 项目概述

### 系统设计
```python
class CustomerServiceBot:
    """智能客服机器人"""
    
    def __init__(self, knowledge_base, llm):
        self.knowledge_base = knowledge_base
        self.llm = llm
        self.conversation_history = []
        self.max_history = 10
    
    def process_query(self, query: str, user_id: str) -> Dict:
        """处理用户查询"""
        
        # 1. 理解用户意图
        intent = self.understand_intent(query)
        
        # 2. 根据意图处理
        if intent == "faq":
            # 常见问题
            response = self.answer_faq(query)
        elif intent == "order":
            # 订单问题
            response = self.check_order(query)
        elif intent == "complaint":
            # 投诉处理
            response = self.handle_complaint(query)
        elif intent == "human":
            # 转人工
            response = self.transfer_to_human()
        else:
            # 通用问答
            response = self.general_qa(query)
        
        # 3. 记录对话
        self.record_conversation(query, response)
        
        return response
    
    def understand_intent(self, query: str) -> str:
        """理解用户意图"""
        
        prompt = f"""
        判断用户查询的意图，返回以下类别之一：
        - faq: 常见问题
        - order: 订单相关
        - complaint: 投诉
        - human: 需要人工
        - other: 其他
        
        用户查询：{query}
        
        意图：
        """
        
        intent = self.llm.generate(prompt).strip().lower()
        return intent
    
    def answer_faq(self, query: str) -> Dict:
        """回答常见问题"""
        
        # 从知识库检索
        result = self.knowledge_base.query(query)
        
        # 检查置信度
        confidence = self.check_confidence(result["answer"])
        
        if confidence < 0.7:
            return {
                "response": "抱歉，我无法准确回答您的问题，正在为您转接人工客服...",
                "need_human": True
            }
        
        return {
            "response": result["answer"],
            "sources": result["sources"],
            "need_human": False
        }
    
    def transfer_to_human(self) -> Dict:
        """转接人工客服"""
        return {
            "response": "正在为您转接人工客服，请稍候...",
            "need_human": True,
            "queue_position": 3  # 队列位置
        }
```
