# 记忆系统

## 短期记忆

### 什么是短期记忆？
**定义**：记住当前对话上下文

### 实现方式
```python
from langchain.memory import ConversationBufferMemory

# 创建记忆
memory = ConversationBufferMemory(
    memory_key="chat_history",
    return_messages=True
)

# 保存对话
memory.save_context(
    {"input": "我叫小明"},
    {"output": "好的，记住了"}
)

# 加载记忆
memory_variables = memory.load_memory_variables({})
print(memory_variables["chat_history"])
```

### 会话缓冲区记忆
```python
from langchain.memory import ConversationBufferWindowMemory

# 只记住最近的K轮对话
memory = ConversationBufferWindowMemory(
    k=5,  # 记住最近5轮
    memory_key="chat_history",
    return_messages=True
)
```

## 长期记忆

### 什么是长期记忆？
**定义**：记住重要信息，跨会话保留

### 实现方式
```python
from langchain.memory import VectorStoreRetrieverMemory
from langchain.vectorstores import FAISS
from langchain.embeddings import OpenAIEmbeddings

# 创建向量存储
embeddings = OpenAIEmbeddings()
vectorstore = FAISS.from_texts(
    ["用户喜欢咖啡", "用户住在北京市"],
    embeddings
)

# 创建长期记忆
memory = VectorStoreRetrieverMemory(
    retriever=vectorstore.as_retriever(),
    memory_key="important_info"
)

# 保存信息
memory.save_context(
    {"input": "我喜欢喝咖啡"},
    {"output": "记住了，你喜欢喝咖啡"}
)

# 检索记忆
relevant_memories = memory.search("用户喜欢什么")
```

## 记忆管理

### 记忆类型
```
1. 短期记忆：当前对话上下文
2. 长期记忆：重要信息，跨会话
3. 情景记忆：特定事件的记忆
4. 语义记忆：通用知识和事实
```

### 记忆管理策略
```python
class MemoryManager:
    def __init__(self):
        self.short_term = ConversationBufferMemory()
        self.long_term = VectorStoreRetrieverMemory()
        
    def add_memory(self, input_text, output_text, importance="normal"):
        """添加记忆"""
        # 保存到短期记忆
        self.short_term.save_context(
            {"input": input_text},
            {"output": output_text}
        )
        
        # 如果重要，保存到长期记忆
        if importance == "high":
            self.long_term.save_context(
                {"input": input_text},
                {"output": output_text}
            )
    
    def retrieve_memory(self, query, memory_type="both"):
        """检索记忆"""
        memories = []
        
        if memory_type in ["both", "short"]:
            short_memories = self.short_term.load_memory_variables({})
            memories.extend(short_memories.get("chat_history", []))
        
        if memory_type in ["both", "long"]:
            long_memories = self.long_term.search(query)
            memories.extend(long_memories)
        
        return memories
```

## 记忆压缩

### 为什么需要压缩？
```
问题：记忆会占用大量空间
解决：压缩重要信息，去除冗余

效果：
  - 减少存储空间
  - 提高检索效率
  - 保持关键信息
```

### 压缩实现
```python
def compress_memory(memory_texts, llm):
    """压缩记忆"""
    
    prompt = f"""
    请将以下对话记忆压缩成关键信息点：
    
    记忆：
    {memory_texts}
    
    要求：
    1. 保留所有重要信息
    2. 去除冗余内容
    3. 使用简洁的语言
    
    关键信息：
    """
    
    compressed = llm.generate(prompt)
    return compressed
```

## 记忆检索

### 相关性检索
```python
def retrieve_relevant_memories(query, memories, top_k=3):
    """检索相关记忆"""
    
    # 计算相似度
    query_embedding = embeddings.embed_query(query)
    
    relevant_memories = []
    for memory in memories:
        memory_embedding = embeddings.embed_query(memory["content"])
        similarity = cosine_similarity(
            query_embedding, 
            memory_embedding
        )
        relevant_memories.append((similarity, memory))
    
    # 排序并返回Top-K
    relevant_memories.sort(reverse=True)
    return [mem for _, mem in relevant_memories[:top_k]]
```

## 多Agent记忆

### 共享记忆
```python
class SharedMemory:
    def __init__(self):
        self.memory_store = {}
        self.access_control = {}
    
    def add_memory(self, agent_id, key, value, shared=False):
        """添加记忆"""
        memory_entry = {
            "value": value,
            "agent": agent_id,
            "timestamp": datetime.now(),
            "shared": shared
        }
        
        self.memory_store[key] = memory_entry
        
        # 设置访问控制
        if shared:
            self.access_control[key] = "all"
        else:
            self.access_control[key] = [agent_id]
    
    def get_memory(self, agent_id, key):
        """获取记忆"""
        if key not in self.memory_store:
            return None
        
        entry = self.memory_store[key]
        
        # 检查访问权限
        if agent_id in self.access_control[key] or \
           self.access_control[key] == "all":
            return entry["value"]
        else:
            return None
```

### 从经验中学习
```python
class LearningMemory:
    def __init__(self):
        self.experiences = []
        self.patterns = {}
    
    def add_experience(self, action, outcome, feedback):
        """添加经验"""
        experience = {
            "action": action,
            "outcome": outcome,
            "feedback": feedback,
            "timestamp": datetime.now()
        }
        
        self.experiences.append(experience)
        
        # 分析模式
        self.analyze_patterns()
    
    def analyze_patterns(self):
        """分析经验模式"""
        # 统计成功和失败的模式
        success_patterns = {}
        failure_patterns = {}
        
        for exp in self.experiences:
            if exp["feedback"] == "positive":
                pattern = extract_pattern(exp["action"], exp["outcome"])
                success_patterns[pattern] = success_patterns.get(pattern, 0) + 1
            else:
                pattern = extract_pattern(exp["action"], exp["outcome"])
                failure_patterns[pattern] = failure_patterns.get(pattern, 0) + 1
        
        self.patterns = {
            "success": success_patterns,
            "failure": failure_patterns
        }
    
    def get_recommendation(self, current_action):
        """获取推荐"""
        # 基于历史经验给出建议
        current_pattern = extract_pattern(current_action, "")
        
        if current_pattern in self.patterns["failure"]:
            return "这个行动之前失败过，建议调整"
        elif current_pattern in self.patterns["success"]:
            return "这个行动之前成功过，可以继续"
        else:
            return "没有相关经验，需要谨慎"
```
