## 客服助手

### 项目概述
```
目标：构建智能客服Agent
功能：自动回答客户问题，处理投诉，转接人工
技术：RAG + Agent + 多轮对话
```

### 系统设计
```python
class CustomerServiceAgent:
    """智能客服Agent"""
    
    def __init__(self, knowledge_base, llm):
        self.knowledge_base = knowledge_base
        self.llm = llm
        self.memory = ConversationBufferMemory()
        self.tools = self._load_tools()
    
    def _load_tools(self):
        """加载工具"""
        return [
            Tool(name="查询订单", func=self.query_order),
            Tool(name="处理退货", func=self.process_return),
            Tool(name="转接人工", func=self.transfer_to_human)
        ]
    
    def respond(self, user_input):
        """响应用户输入"""
        
        # 1. 理解用户意图
        intent = self._understand_intent(user_input)
        
        # 2. 根据意图选择处理方式
        if intent == "faq":
            response = self._answer_faq(user_input)
        elif intent == "order":
            response = self._handle_order(user_input)
        elif intent == "complaint":
            response = self._handle_complaint(user_input)
        else:
            response = self._general_response(user_input)
        
        # 3. 保存对话历史
        self.memory.save_context(
            {"input": user_input},
            {"output": response}
        )
        
        return response
    
    def _understand_intent(self, text):
        """理解用户意图"""
        prompt = f"""
        判断用户意图，返回以下之一：
        - faq: 常见问题
        - order: 订单相关
        - complaint: 投诉
        - other: 其他
        
        用户输入：{text}
        
        意图：
        """
        
        return self.llm.generate(prompt).strip().lower()
```

## 代码助手

### 项目概述
```
目标：构建代码生成Agent
功能：代码生成、调试、解释、优化
技术：代码理解 + 工具调用 + 迭代改进
```

### 代码实现
```python
class CodeAssistantAgent:
    """代码助手Agent"""
    
    def __init__(self, llm):
        self.llm = llm
        self.tools = [
            Tool(name="执行代码", func=self.execute_code),
            Tool(name="运行测试", func=self.run_tests),
            Tool(name="代码审查", func=self.code_review)
        ]
    
    def generate_code(self, requirement):
        """生成代码"""
        
        # 1. 理解需求
        analysis = self._analyze_requirement(requirement)
        
        # 2. 生成初始代码
        code = self._generate_initial_code(analysis)
        
        # 3. 测试和改进
        for i in range(3):  # 最多迭代3次
            test_result = self._test_code(code)
            
            if test_result["success"]:
                break
            
            # 根据测试结果改进代码
            code = self._improve_code(code, test_result["errors"])
        
        return code
    
    def debug_code(self, code, error_message):
        """调试代码"""
        
        # 1. 分析错误
        error_analysis = self._analyze_error(code, error_message)
        
        # 2. 生成修复方案
        fix = self._generate_fix(code, error_analysis)
        
        return fix
    
    def explain_code(self, code):
        """解释代码"""
        
        prompt = f"""
        请解释以下代码的功能和工作原理：
        
        ```python
        {code}
        ```
        
        解释：
        """
        
        return self.llm.generate(prompt)
```

功能：文献搜索、数据分析、报告生成
技术：RAG + 数据分析 + 报告生成
```

### 代码实现
```python
class ResearchAssistantAgent:
    """研究助手Agent"""
    
    def __init__(self, llm, vectorstore):
        self.llm = llm
        self.vectorstore = vectorstore
        self.tools = [
            Tool(name="搜索文献", func=self.search_papers),
            Tool(name="分析数据", func=self.analyze_data),
            Tool(name="生成图表", func=self.generate_chart)
        ]
    
    def research_topic(self, topic, depth="medium"):
        """研究主题"""
        
        # 1. 文献搜索
        papers = self.search_papers(topic, limit=10)
        
        # 2. 阅读和总结
        summaries = []
        for paper in papers:
            summary = self._summarize_paper(paper)
            summaries.append(summary)
        
        # 3. 综合分析
        analysis = self._synthesize_analysis(summaries)
        
        # 4. 生成报告
        report = self._generate_report(topic, analysis)
        
        return report
    
    def compare_methods(self, methods):
        """比较方法"""
        
        prompt = f"""
        请比较以下方法：
        {methods}
        
        比较维度：
        1. 原理
        2. 优缺点
        3. 适用场景
        4. 性能
        
        比较结果：
        """
        
        return self.llm.generate(prompt)
```

### 主程序
```python
# main.py
import yaml
from agents.customer_service import CustomerServiceAgent
from agents.code_assistant import CodeAssistantAgent
from agents.research_assistant import ResearchAssistantAgent

def main():
    # 加载配置
    with open("config/config.yaml", "r") as f:
        config = yaml.safe_load(f)
    
    # 初始化LLM
    llm = ChatOpenAI(
        model=config["llm"]["model"],
        temperature=config["llm"]["temperature"]
    )
    
    # 初始化向量数据库
    vectorstore = init_vectorstore(config["vectorstore"])
    
    # 创建Agent
    agents = {
        "customer_service": CustomerServiceAgent(llm, vectorstore),
        "code_assistant": CodeAssistantAgent(llm),
        "research_assistant": ResearchAssistantAgent(llm, vectorstore)
    }
    
    # 运行Agent
    while True:
        print("\n可用Agent：")
        for name in agents.keys():
            print(f"  - {name}")
        
        agent_name = input("\n选择Agent：")
        if agent_name not in agents:
            print("无效选择")
            continue
        
        agent = agents[agent_name]
        
        print(f"\n使用 {agent_name}（输入'quit'退出）")
        while True:
            user_input = input("\n你：")
            if user_input.lower() == 'quit':
                break
            
            response = agent.respond(user_input)
            print(f"\n助手：{response}")

if __name__ == "__main__":
    main()
```

## 评估方法

### Agent评估框架
```python
class AgentEvaluator:
    """Agent评估器"""
    
    def __init__(self, test_cases):
        self.test_cases = test_cases
    
    def evaluate(self, agent):
        """评估Agent"""
        
        results = {
            "accuracy": 0,
            "response_time": 0,
            "satisfaction": 0,
            "errors": []
        }
        
        for test_case in self.test_cases:
            # 测试Agent
            start_time = time.time()
            response = agent.respond(test_case["input"])
            end_time = time.time()
            
            # 评估响应
            accuracy = self._evaluate_accuracy(
                response, 
                test_case["expected"]
            )
            
            results["accuracy"] += accuracy
            results["response_time"] += (end_time - start_time)
            
            if accuracy < 0.8:
                results["errors"].append({
                    "input": test_case["input"],
                    "response": response,
                    "expected": test_case["expected"]
                })
        
        # 计算平均值
        num_cases = len(self.test_cases)
        results["accuracy"] /= num_cases
        results["response_time"] /= num_cases
        
        return results
```
