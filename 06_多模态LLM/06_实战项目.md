# 实战项目

## 图像助手

### 项目概述
```
目标：构建图像理解和分析助手
功能：图像描述、问答、编辑建议
技术：多模态模型 + RAG
```

### 代码实现
```python
class ImageAssistant:
    """图像助手"""
    
    def __init__(self):
        from transformers import LlavaNextProcessor, LlavaNextForConditionalGeneration
        
        self.processor = LlavaNextProcessor.from_pretrained(
            "llava-hf/llava-v1.6-mistral-7b-hf"
        )
        self.model = LlavaNextForConditionalGeneration.from_pretrained(
            "llava-hf/llava-v1.6-mistral-7b-hf",
            torch_dtype=torch.float16,
            device_map="auto"
        )
    
    def describe_image(self, image_path):
        """描述图像"""
        
        from PIL import Image
        
        image = Image.open(image_path)
        prompt = "[INST] <image>\n请详细描述这张图片 [/INST]"
        
        inputs = self.processor(
            prompt, 
            image, 
            return_tensors="pt"
        ).to("cuda")
        
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=200
            )
        
        return self.processor.decode(
            outputs[0], 
            skip_special_tokens=True
        )
    
    def answer_question(self, image_path, question):
        """回答图像相关问题"""
        
        from PIL import Image
        
        image = Image.open(image_path)
        prompt = f"[INST] <image>\n{question} [/INST]"
        
        inputs = self.processor(
            prompt, 
            image, 
            return_tensors="pt"
        ).to("cuda")
        
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=100
            )
        
        return self.processor.decode(
            outputs[0], 
            skip_special_tokens=True
        )
```

## 视频分析器

### 项目概述
```
目标：构建视频内容分析器
功能：视频摘要、关键帧提取、内容理解
技术：视频理解 + 多模态分析
```

### 代码实现
```python
class VideoAnalyzer:
    """视频分析器"""
    
    def __init__(self):
        self.image_analyzer = ImageAssistant()
    
    def extract_key_frames(self, video_path, num_frames=10):
        """提取关键帧"""
        import cv2
        
        cap = cv2.VideoCapture(video_path)
        frames = []
        
        # 获取总帧数
        total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
        interval = total_frames // num_frames
        
        for i in range(0, total_frames, interval):
            cap.set(cv2.CAP_PROP_POS_FRAMES, i)
            ret, frame = cap.read()
            
            if ret:
                # 转换为PIL图像
                from PIL import Image
                frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                frames.append(Image.fromarray(frame_rgb))
        
        cap.release()
        return frames
    
    def analyze_video(self, video_path):
        """分析视频内容"""
        
        # 1. 提取关键帧
        frames = self.extract_key_frames(video_path)
        
        # 2. 分析每帧
        analyses = []
        for i, frame in enumerate(frames):
            # 保存帧
            frame_path = f"frame_{i}.jpg"
            frame.save(frame_path)
            
            # 分析帧
            analysis = self.image_analyzer.describe_image(frame_path)
            analyses.append({
                "frame": i,
                "timestamp": i * 5,  # 假设每5秒一帧
                "analysis": analysis
            })
        
        # 3. 生成视频摘要
        summary = self.generate_video_summary(analyses)
        
        return {
            "frames": analyses,
            "summary": summary
        }
    
    def generate_video_summary(self, analyses):
        """生成视频摘要"""
        
        prompt = f"""
        以下是视频的关键帧分析：
        {analyses}
        
        请生成一个连贯的视频摘要：
        """
        
        from langchain.chat_models import ChatOpenAI
        llm = ChatOpenAI(temperature=0.7)
        
        return llm.generate(prompt)
```

## 多模态助手

### 项目概述
```
目标：构建全能多模态助手
功能：图像、语音、视频处理
技术：多模态模型集成
```

### 代码实现
```python
class MultimodalAssistant:
    """多模态助手"""
    
    def __init__(self):
        self.image_assistant = ImageAssistant()
        self.video_analyzer = VideoAnalyzer()
        # 语音处理可以使用Whisper等
    
    def process_image(self, image_path, task="describe", **kwargs):
        """处理图像任务"""
        
        if task == "describe":
            return self.image_assistant.describe_image(image_path)
        elif task == "question":
            question = kwargs.get("question", "这是什么？")
            return self.image_assistant.answer_question(image_path, question)
        else:
            return "不支持的任务类型"
    
    def process_video(self, video_path, task="analyze"):
        """处理视频任务"""
        
        if task == "analyze":
            return self.video_analyzer.analyze_video(video_path)
        elif task == "summary":
            analysis = self.video_analyzer.analyze_video(video_path)
            return analysis["summary"]
        else:
            return "不支持的任务类型"
    
    def process_audio(self, audio_path, task="transcribe"):
        """处理音频任务"""
        
        if task == "transcribe":
            # 使用Whisper转录音频
            import whisper
            model = whisper.load_model("base")
            result = model.transcribe(audio_path)
            return result["text"]
        else:
            return "不支持的任务类型"
```
