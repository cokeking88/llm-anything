# 图像理解

## 图像编码器

### 什么是图像编码器？
**定义**：将图像转换为向量表示的模型

### 主流图像编码器
```
1. Vision Transformer (ViT)
   - 将图像分割为patch
   - 使用Transformer处理
   - 效果好，计算量大

2. CLIP
   - OpenAI开发
   - 联合训练图像和文本
   - 支持零样本分类

3. ResNet
   - CNN架构
   - 经典模型
   - 轻量高效

4. DINOv2
   - 自监督学习
   - 通用特征提取
   - 效果优秀
```

### 使用CLIP
```python
from transformers import CLIPProcessor, CLIPModel
from PIL import Image
import torch

# 加载模型
model = CLIPModel.from_pretrained("openai/clip-vit-base-patch32")
processor = CLIPProcessor.from_pretrained("openai/clip-vit-base-patch32")

# 加载图像
image = Image.open("image.jpg")

# 文本描述
texts = ["a photo of a cat", "a photo of a dog", "a photo of a car"]

# 处理输入
inputs = processor(text=texts, images=image, return_tensors="pt", padding=True)

# 计算相似度
with torch.no_grad():
    outputs = model(**inputs)
    logits_per_image = outputs.logits_per_image
    probs = logits_per_image.softmax(dim=1)

print(probs)
# [[0.95, 0.03, 0.02]] -> 95%概率是猫
```

## 视觉问答（VQA）

### 什么是视觉问答？
**定义**：基于图像回答问题

### VQA实现
```python
from transformers import LlavaNextProcessor, LlavaNextForConditionalGeneration
from PIL import Image
import torch

# 加载模型
processor = LlavaNextProcessor.from_pretrained("llava-hf/llava-v1.6-mistral-7b-hf")
model = LlavaNextForConditionalGeneration.from_pretrained(
    "llava-hf/llava-v1.6-mistral-7b-hf",
    torch_dtype=torch.float16,
    device_map="auto"
)

# 加载图像
image = Image.open("image.jpg")

# 提问
question = "这张图片里有什么？"

# 处理输入
prompt = f"[INST] <image>\n{question} [/INST]"
inputs = processor(prompt, image, return_tensors="pt").to("cuda")

# 生成回答
with torch.no_grad():
    outputs = model.generate(**inputs, max_new_tokens=100)

answer = processor.decode(outputs[0], skip_special_tokens=True)
print(answer)
```

### 批量VQA
```python
def batch_vqa(images, questions, model, processor):
    """批量视觉问答"""
    answers = []
    
    for image, question in zip(images, questions):
        prompt = f"[INST] <image>\n{question} [/INST]"
        inputs = processor(prompt, image, return_tensors="pt").to("cuda")
        
        with torch.no_grad():
            outputs = model.generate(**inputs, max_new_tokens=100)
        
        answer = processor.decode(outputs[0], skip_special_tokens=True)
        answers.append(answer)
    
    return answers
```

### 零样本分类（CLIP）
```python
from transformers import CLIPProcessor, CLIPModel
from PIL import Image
import torch

# 加载模型
model = CLIPModel.from_pretrained("openai/clip-vit-base-patch32")
processor = CLIPProcessor.from_pretrained("openai/clip-vit-base-patch32")

# 加载图像
image = Image.open("image.jpg")

# 定义类别
candidate_labels = ["a cat", "a dog", "a bird", "a car"]

# 处理输入
inputs = processor(
    text=candidate_labels,
    images=image,
    return_tensors="pt",
    padding=True
)

# 计算概率
with torch.no_grad():
    outputs = model(**inputs)
    logits_per_image = outputs.logits_per_image
    probs = logits_per_image.softmax(dim=1)

# 输出结果
for label, prob in zip(candidate_labels, probs[0]):
    print(f"{label}: {prob.item():.2%}")
```

## 目标检测

### 使用DETR
```python
from transformers import DetrImageProcessor, DetrForObjectDetection
from PIL import Image
import torch

# 加载模型
processor = DetrImageProcessor.from_pretrained("facebook/detr-resnet-50")
model = DetrForObjectDetection.from_pretrained("facebook/detr-resnet-50")

# 加载图像
image = Image.open("image.jpg")

# 处理输入
inputs = processor(images=image, return_tensors="pt")

# 检测
with torch.no_grad():
    outputs = model(**inputs)

# 后处理
target_sizes = torch.tensor([image.size[::-1]])
results = processor.post_process_object_detection(
    outputs, target_sizes=target_sizes, threshold=0.9
)[0]

# 输出检测结果
for score, label, box in zip(results["scores"], results["labels"], results["boxes"]):
    print(f"检测到: {model.config.id2label[label.item()]} "
          f"置信度: {score.item():.2%} "
          f"位置: {box.tolist()}")
```

## 多模态特征提取

### CLIP特征提取
```python
def extract_clip_features(images, texts, model, processor):
    """提取CLIP特征"""
    
    # 图像特征
    image_inputs = processor(
        images=images,
        return_tensors="pt",
        padding=True
    ).to("cuda")
    
    with torch.no_grad():
        image_features = model.get_image_features(**image_inputs)
    
    # 文本特征
    text_inputs = processor(
        text=texts,
        return_tensors="pt",
        padding=True
    ).to("cuda")
    
    with torch.no_grad():
        text_features = model.get_text_features(**text_inputs)
    
    # 归一化
    image_features = image_features / image_features.norm(dim=-1, keepdim=True)
    text_features = text_features / text_features.norm(dim=-1, keepdim=True)
    
    return image_features, text_features
```

## 性能优化

### 1. 批量处理
```python
def process_images_batch(images, model, processor, batch_size=32):
    """批量处理图像"""
    results = []
    
    for i in range(0, len(images), batch_size):
        batch = images[i:i+batch_size]
        inputs = processor(
            images=batch,
            return_tensors="pt",
            padding=True
        ).to("cuda")
        
        with torch.no_grad():
            outputs = model(**inputs)
        
        results.append(outputs)
    
    return results
```

### 2. 缓存机制
```python
import hashlib
import pickle
import os

class FeatureCache:
    def __init__(self, cache_dir="./cache"):
        self.cache_dir = cache_dir
        os.makedirs(cache_dir, exist_ok=True)
    
    def get_cache_key(self, image_path):
        """计算缓存键"""
        with open(image_path, 'rb') as f:
            image_hash = hashlib.md5(f.read()).hexdigest()
        return image_hash
    
    def get(self, image_path):
        """获取缓存"""
        cache_key = self.get_cache_key(image_path)
        cache_file = os.path.join(self.cache_dir, f"{cache_key}.pkl")
        
        if os.path.exists(cache_file):
            with open(cache_file, 'rb') as f:
                return pickle.load(f)
        return None
    
    def set(self, image_path, features):
        """设置缓存"""
        cache_key = self.get_cache_key(image_path)
        cache_file = os.path.join(self.cache_dir, f"{cache_key}.pkl")
        
        with open(cache_file, 'wb') as f:
            pickle.dump(features, f)
```
