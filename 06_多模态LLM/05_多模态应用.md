# 多模态应用

## 视觉问答系统

### 什么是视觉问答？
**定义**：基于图像回答问题

### 系统架构
```
输入：图像 + 问题
处理：多模态模型理解
输出：答案
```

### 完整系统示例
```python
class VisualQASystem:
    """视觉问答系统"""
    
    def __init__(self, model_name="llava-hf/llava-v1.6-mistral-7b-hf"):
        from transformers import LlavaNextProcessor, LlavaNextForConditionalGeneration
        
        self.processor = LlavaNextProcessor.from_pretrained(model_name)
        self.model = LlavaNextForConditionalGeneration.from_pretrained(
            model_name,
            torch_dtype=torch.float16,
            device_map="auto"
        )
    
    def answer_question(self, image_path, question):
        """回答图像相关问题"""
        
        from PIL import Image
        
        # 加载图像
        image = Image.open(image_path)
        
        # 准备输入
        prompt = f"[INST] <image>\n{question} [/INST]"
        inputs = self.processor(
            prompt, 
            image, 
            return_tensors="pt"
        ).to("cuda")
        
        # 生成回答
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=100,
                temperature=0.7
            )
        
        answer = self.processor.decode(
            outputs[0], 
            skip_special_tokens=True
        )
        
        return answer
```

## 多模态搜索

### 什么是多模态搜索？
**定义**：基于多种模态（图像、文本等）的搜索

### 图像搜索系统
```python
class MultimodalSearch:
    """多模态搜索系统"""
    
    def __init__(self):
        from transformers import CLIPProcessor, CLIPModel
        
        self.model = CLIPModel.from_pretrained("openai/clip-vit-base-patch32")
        self.processor = CLIPProcessor.from_pretrained("openai/clip-vit-base-patch32")
        self.image_index = {}
    
    def add_image(self, image_id, image_path):
        """添加图像到索引"""
        
        from PIL import Image
        
        image = Image.open(image_path)
        inputs = self.processor(
            images=image,
            return_tensors="pt",
            padding=True
        )
        
        with torch.no_grad():
            image_features = self.model.get_image_features(**inputs)
        
        self.image_index[image_id] = {
            "features": image_features,
            "path": image_path
        }
    
    def search_by_text(self, query, top_k=5):
        """文本搜索图像"""
        
        # 文本编码
        text_inputs = self.processor(
            text=query,
            return_tensors="pt",
            padding=True
        )
        
        with torch.no_grad():
            text_features = self.model.get_text_features(**text_inputs)
        
        # 计算相似度
        results = []
        for image_id, data in self.image_index.items():
            similarity = torch.cosine_similarity(
                text_features,
                data["features"]
            ).item()
            
            results.append({
                "image_id": image_id,
                "path": data["path"],
                "similarity": similarity
            })
        
        # 排序并返回Top-K
        results.sort(key=lambda x: x["similarity"], reverse=True)
        return results[:top_k]
```
